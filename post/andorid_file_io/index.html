Arm

 arm:   bionic/libc/arch-arm/syscalls/__openat.S

ENTRY(__openat)
    mov     ip, r7
    ldr     r7, =__NR_openat
    swi     #0
    mov     r7, ip
    cmn     r0, #(MAX_ERRNO + 1)
    bxls    lr
    neg     r0, r0
    b       __set_errno_internal
END(__openat)

X86

x86:  bionic/libc/arch-x86/syscalls/__openat.S

ENTRY(__openat)
    pushl   %ebx
    pushl   %ecx
    pushl   %edx
    pushl   %esi
    mov     20(%esp), %ebx
    mov     24(%esp), %ecx
    mov     28(%esp), %edx
    mov     32(%esp), %esi
    movl    $__NR_openat, %eax
    int     $0x80
    cmpl    $-MAX_ERRNO, %eax
    jb      1f
    negl    %eax
    pushl   %eax
    call    __set_errno_internal
    addl    $4, %esp
1:
    popl    %esi
    popl    %edx
    popl    %ecx
    popl    %ebx
    ret
END(__openat)
